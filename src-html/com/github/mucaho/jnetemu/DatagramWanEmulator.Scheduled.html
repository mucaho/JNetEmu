033/**
034 * A convenience class for debugging UDP communication. The DatagramWanEmulator sits between
035 * two specified {@link java.net.DatagramSocket}s and emulates various network conditions.
036 * <p>
037 * In order to use the emulator, you send the {@link java.net.DatagramPacket}s directly
038 * to the emulator's {@link java.net.SocketAddress} instead of sending them to the receiver
039 * {@link java.net.SocketAddress}.
040 * The emulator will then apply various network conditions,
041 * before sending those packets to the receiver {@link java.net.SocketAddress}.
042 * <p>
043 * This class can be partially instantiated using the respective
044 * {@link com.github.mucaho.jnetemu.DatagramWanEmulator#DatagramWanEmulator(SocketAddress, SocketAddress, SocketAddress) constructor}.
045 * Use the appropriate methods to {@link com.github.mucaho.jnetemu.DatagramWanEmulator#startEmulation() start} and
046 * {@link com.github.mucaho.jnetemu.DatagramWanEmulator#stopEmulation() stop} the emulation.
047 * The actual computation of {@link #computeNetworkConditions(long, Collection, List) network conditions}
048 * is given by subclasses of this abstract class.
049 * <p>
050 * Internally, this emulator uses a single thread for all emulator instances
051 * by leveraging {@link java.nio NIO} selectors &amp; non-blocking channels.
052 * <p>
053 * There are multiple emulator configuration options available,
054 * some possibilities are illustrated in the following figures:
055 * <br>
056 * Emulator sitting on PeerA, emulating network conditions for outgoing traffic only.
057 * <pre>{@literal
058 * ╔══════════╗      ╔═════╗      ╔═════════╗
059 * ║ A -> EMU ║ ---> ║     ║ ---> ║    A    ║
060 * ║ P        ║      ║ ??? ║      ║    P    ║
061 * ║ P  <--   ║ <--- ║     ║ <--- ║    P    ║
062 * ╚══════════╝      ╚═════╝      ╚═════════╝
063 *    PeerA          Network         PeerB
064 * }</pre>
065 * <br>
066 * Emulator sitting on PeerA, emulating network conditions for both outgoing and incoming traffic.
067 * <pre>{@literal
068 * ╔═════════╗      ╔═════╗      ╔═════════╗
069 * ║ A     E ║      ║     ║      ║    A    ║
070 * ║ P <-> M ║ <--> ║ ??? ║ <--> ║    P    ║
071 * ║ P     U ║      ║     ║      ║    P    ║
072 * ╚═════════╝      ╚═════╝      ╚═════════╝
073 *    PeerA         Network         PeerB
074 * }</pre>
075 * <br>
076 * One emulator sitting on each peer, emulating network conditions for its outgoing traffic respectively.
077 * <pre>{@literal
078 * ╔══════════╗      ╔═════╗      ╔══════════╗
079 * ║ A -> EMU ║ ---> ║     ║ ---> ║    --> A ║
080 * ║ P        ║      ║ ??? ║      ║        P ║
081 * ║ P <--    ║ <--- ║     ║ <--- ║ EMU <- P ║
082 * ╚══════════╝      ╚═════╝      ╚══════════╝
083 *    PeerA          Network         PeerB
084 * }</pre>
085 *
086 */
087public abstract class DatagramWanEmulator {

093    /**
094     * Minimum payload size in bytes of an UDP datagram that will not cause fragmentation in the IP layer.
095     * <p>
096     * That's the minimum supported MTU for IPv4 ({@code 576} B), minus the maximum IPv4 header size ({@code 60} B)
097     * and minus the UDP header size ({@code 8} B). Defaults to {@value} B.
098     */
099    public final static int MINIMUM__PACKET_SIZE = 576 - 60 - 8;
100    /**
101     * Size in bytes of the usually supported MTU for IPv4. Defaults to {@value} B.
102     */
103    public final static int DEFAULT_PACKET_SIZE = 1500;
104    // incremented on instance creation; decremented on instance destruction
105    private final static AtomicInteger emulatorCount = new AtomicInteger(0);
106    // setup on first instance creation; torn-down on last instance destruction
107    private static Thread selectionThread;
108    private static Selector selector;

133    /**
134     * Construct a new {@code DatagramWanEmulator}. This will automatically create a new socket
135     * bound to the specified {@code emulatorAddress},
136     * with {@code maxPacketSize} set to {@link #MINIMUM__PACKET_SIZE MINIMUM__PACKET_SIZE}.
137     *
138     * @param emulatorAddress the emulator address
139     * @param socketAddressA  one socket address
140     * @param socketAddressB  the other socket address
141     */
142    public DatagramWanEmulator(SocketAddress emulatorAddress, SocketAddress socketAddressA,
143                               SocketAddress socketAddressB) {
144        this(emulatorAddress, socketAddressA, socketAddressB, MINIMUM__PACKET_SIZE);
145    }
146
147    /**
148     * Construct a new {@code DatagramWanEmulator}. This will automatically create a new socket
149     * bound to the specified {@code emulatorAddress}.
150     *
151     * @param emulatorAddress the emulator address
152     * @param socketAddressA  one socket address
153     * @param socketAddressB  the other socket address
154     * @param maxPacketSize the maximum size of incoming / outgoing datagram packets (in bytes).
155     */
156    public DatagramWanEmulator(SocketAddress emulatorAddress, SocketAddress socketAddressA,
157                               SocketAddress socketAddressB, int maxPacketSize) {
158        this.socketAddressA = socketAddressA;
159        this.socketAddressB = socketAddressB;
160        this.emulatorAddress = emulatorAddress;
161        this.maxPacketSize = maxPacketSize;
162
163        allocateBuffers(16);
164    }
165
166    /**
167     * Expand the amount of available buffers to write to.
168     */
169    private void allocateBuffers(int amount) {
170        do {
171            ByteBuffer buffer = ByteBuffer.allocateDirect(maxPacketSize);
172            writableBuffers.offer(buffer);
173        } while (--amount > 0);
174    }

181    /**
182     * Compute network conditions. Actual implementation given in subclasses.
183     * <br>
184     * This method will be called once per incoming packet and computes the network conditions for the outgoing packet.
185     * At any time it is guaranteed to be called from a single thread only,
186     * thus implementing classes should not worry about threading issues.
187     *
188     * @param timeNow the current low-resolution system time in {@code ms},
189     *                that should be used by all time-dependent computations.
190     *                It represents the difference between the current time
191     *                and midnight, 01.01.1970 UTC.
192     * @param scheduled a read-only ordered collection containing already scheduled packets.
193     *                  Note that packets being duplicated are represented multiple times in this collection.
194     *                  This is provided as additional information for computing network conditions.
195     * @param scheduledTimes a writable list that should be filled with zero or more scheduled times
196     *                       at which the incoming packet will be relayed.
197     *                       Scheduled times are given as the difference between current time ({@code timeNow})
198     *                       and midnight, 01.01.1970 UTC.
199     *                       <ul>
200     *                       <li>A zero-length list indicates packet gets dropped.</li>
201     *                       <li>A list with a single element indicates at which time to relay the incoming packet.</li>
202     *                       <li>A list with {@code n} elements indicates that the packet will be duplicated {@code n-1} times.
203     *                       These duplicates will be sent according to the time values provided.</li>
204     *                       </ul>
205     */
206    protected abstract void computeNetworkConditions(long timeNow,
207                                                     Collection<? extends Scheduled> scheduled,
208                                                     List<Long> scheduledTimes);
209
210    private final List<Long> scheduledTimes = new ArrayList<Long>();
211    private final Collection<? extends Scheduled> scheduledOut = Collections.unmodifiableCollection(scheduledPackets);
212    private List<Long> computeNetworkConditions(long timeNow) {
213        scheduledTimes.clear();
214        computeNetworkConditions(timeNow, scheduledOut, scheduledTimes);
215        return scheduledTimes;
216    }
217
218    /**
219     * Read from channel to buffers.
220     * @param timeNow current time in ms
221     * @return {@code true} if something was read, {@code false} otherwise
222     */
223    private boolean read(long timeNow) throws IOException {
224        ByteBuffer buffer = writableBuffers.poll();
225        if (buffer == null) {
226            allocateBuffers(16);
227            buffer = writableBuffers.poll();
228        }
229
230        SocketAddress senderAddress = datagramChannel.receive(buffer);
231        if (senderAddress == null) { // if there was no datagram to receive yet
232            buffer.clear();
233            writableBuffers.offer(buffer);
234            return false;
235        }
236        buffer.flip();
237
238        SocketAddress receiverAddress;
239        if (socketAddressA.equals(senderAddress))
240            receiverAddress = socketAddressB;
241        else if (socketAddressB.equals(senderAddress))
242            receiverAddress = socketAddressA;
243        else
244            receiverAddress = null;
245        if (receiverAddress == null) { // if datagram was received from 3rd party
246            buffer.clear();
247            writableBuffers.offer(buffer);
248            return true;
249        }
250
251        PacketId packetId = new PacketId();
252        for (Long scheduledTime : computeNetworkConditions(timeNow)) {
253            packetId.count++;
254            ScheduledPacket scheduled = new ScheduledPacket(buffer, receiverAddress, packetId, scheduledTime);
255            scheduledPackets.offer(scheduled);
256        }
257
258        if (packetId.count == 0) {
259            buffer.clear();
260            writableBuffers.offer(buffer);
261        }
262
263        return true;
264    }
265
266    /**
267     * Check if there are buffers ready to be written.
268     * @param timeNow the current time in ms
269     * @return {@code true} if something was written, {@code false} otherwise
270     */
271    private boolean needsWriting(long timeNow) {
272        ScheduledPacket scheduled = scheduledPackets.peek();
273        return scheduled != null && scheduled.isReady(timeNow);
274    }
275
276    /**
277     * Write from ready buffers to channel.
278     * @param timeNow current time in ms
279     * @return {@code true} if something was written, {@code false} otherwise
280     */
281    private boolean write(long timeNow) throws IOException {
282        ScheduledPacket scheduled = scheduledPackets.peek();
283        if (scheduled == null || !scheduled.isReady(timeNow)) { // if there is no ready packet yet
284            return false;
285        }
286        scheduled = scheduledPackets.poll();
287        ByteBuffer buffer = scheduled.buffer;
288
289        buffer.rewind();
290        int bytesSent = datagramChannel.send(buffer, scheduled.receiverAddress);
291        if (bytesSent <= 0) { // if datagram could not be sent yet
292            buffer.rewind();
293            scheduledPackets.offer(scheduled);
294            return false;
295        }
296
297        if (--scheduled.packetId.count == 0) {
298            buffer.clear();
299            writableBuffers.offer(buffer);
300        }
301
302        return true;
303    }

308    /// SELECTION THREAD ///
309    ////////////////////////
310
311    private final static Runnable runSelection = new Runnable() {
312        @Override
313        public void run() {
314            try {
315                // run the selection loop until interrupted
316                while(!Thread.currentThread().isInterrupted()) {
317                    long timeNow = System.currentTimeMillis();
318
319                    try {
320                        // check which buffers are ready to be written
321                        for (SelectionKey key : selector.keys()) {
322                            DatagramWanEmulator emulator = (DatagramWanEmulator) key.attachment();
323                            if (emulator.needsWriting(timeNow))
324                                // add OP_WRITE as interestOp
325                                key.interestOps(key.interestOps() | SelectionKey.OP_WRITE);
326                        }
327
328                        // do reading and writing on ready channels
329                        selector.selectNow();
330                        Iterator<SelectionKey> keyIterator = selector.selectedKeys().iterator();
331                        while (keyIterator.hasNext()) {
332                            SelectionKey key = keyIterator.next();
333
334                            DatagramWanEmulator emulator = (DatagramWanEmulator) key.attachment();
335                            if (key.isReadable()) {
336                                while (emulator.read(timeNow)) ;
337                            }
338                            if (key.isWritable()) {
339                                while (emulator.write(timeNow)) ;
340                                // remove OP_WRITE as interestOp
341                                key.interestOps(key.interestOps() & ~SelectionKey.OP_WRITE);
342                            }
343
344                            keyIterator.remove();
345                        }
346                    } catch (CancelledKeyException e) {
347                        // key cancelled from other thread; safe to ignore
348                    } catch (ClosedChannelException e) {
349                        // channel closed from other thread; safe to ignore
350                    }
351
352                    Thread.yield();
353                }
354            } catch (IOException e) {
355                // some not-handleable error occurred
356                e.printStackTrace();
357            }
358        }
359    };

366    /**
367     * Start the emulation between the two sockets.
368     *
369     * @throws ClosedChannelException if the current instance has already been stopped by
370     * {@link #stopEmulation() stopEmulation()}. Create a new
371     * {@link com.github.mucaho.jnetemu.DatagramWanEmulator#DatagramWanEmulator(SocketAddress, SocketAddress, SocketAddress) instance} instead.
372     * @throws java.io.IOException if there was some other I/O error
373     */
374    public void startEmulation() throws IOException {
375        if (datagramChannel != null && !datagramChannel.isOpen())
376            throw new ClosedChannelException();
377
378        // setup up a new selectionThread if this is the first emulator instance
379        synchronized (DatagramWanEmulator.emulatorCount) {
380            if (emulatorCount.getAndIncrement() == 0) {
381                // open a selector
382                selector = Selector.open();
383
384                selectionThread = new Thread(runSelection);
385                selectionThread.start();
386            }
387        }
388
389        // register this instance to the selectionThread
390        datagramChannel = DatagramChannel.open();
391        datagramChannel.configureBlocking(false);
392        datagramChannel.socket().bind(emulatorAddress);
393        selectionKey = datagramChannel.register(selector, SelectionKey.OP_READ, this);
394    }
395
396    /**
397     * Stop the emulation between the two sockets and close the associated emulator socket.
398     *
399     * @throws java.lang.InterruptedException if the current instance is interrupted while shutting down last emulation
400     * @throws java.io.IOException if there was some other I/O error
401     */
402    public void stopEmulation() throws InterruptedException, IOException {
403        // destroy the old selectionThread if this is was the last emulator instance
404        synchronized (DatagramWanEmulator.emulatorCount) {
405            if (emulatorCount.decrementAndGet() == 0) {
406                selectionThread.interrupt();
407                selectionThread.join();
408                selectionThread = null;
409
410                // close the selector
411                selector.close();
412                selector = null;
413            }
414        }
415
416        // deregister this instance from the selectionThread
417        selectionKey.cancel();
418        datagramChannel.close();
419    }

427    private final static class PacketId {
428        private int count = 0;
429    }

431    /**
432     * Interface for handling scheduled objects, that should be processed at a given time.
433     * <br>
434     * The time is represented as the difference in {@code ms},
435     * between the time this object is ready to be processed and midnight, 01.01.1970 UTC.
436     */
437    public interface Scheduled extends Comparable<Scheduled> {
438        /**
439         * Retrieve the time this object will be ready to be processed.
440         * @return the difference in {@code ms}, between the time this object is ready to be processed
441         *         and midnight, 01.01.1970 UTC.
442         */
443        long getScheduledTime();
444
445        /**
446         * Check whether this object is ready to be processed.
447         * @param timeNow the current low-resolution system time in {@code ms},
448         *                which will be used to determine if this object is ready yet.
449         *                It represents the difference between the current time
450         *                and midnight, 01.01.1970 UTC.
451         * @return {@code true}, iff object is ready to be processed, {@code false} otherwise
452         */
453        boolean isReady(long timeNow);
454    }
455
456    private final static class ScheduledPacket implements Scheduled {
457        private final ByteBuffer buffer;
458        private final SocketAddress receiverAddress;
459        private final long finishedTime;
460        private final PacketId packetId;
461
462        private ScheduledPacket(ByteBuffer buffer, SocketAddress receiverAddress,
463                                PacketId packetId, long finishedTime) {
464            this.buffer = buffer;
465            this.receiverAddress = receiverAddress;
466            this.finishedTime = finishedTime;
467            this.packetId = packetId;
468        }
469
470        @Override
471        public long getScheduledTime() {
472            return finishedTime;
473        }
474
475        @Override
476        public boolean isReady(long timeNow) {
477            return finishedTime <= timeNow;
478        }
479
480        @Override
481        public int compareTo(Scheduled other) {
482            long thisDelay = this.getScheduledTime();
483            long otherDelay = other.getScheduledTime();
484            return (int) (thisDelay - otherDelay);
485        }
486    }
