024package com.github.mucaho.jnetemu;
025
026
027import java.net.SocketAddress;
028import java.util.Collection;
029import java.util.List;
030import java.util.Random;
031
032/**
033 * This class provides a simple implementation of a {@link DatagramWanEmulator network emulator}.
034 * It introduces packet-latency, -jitter, -loss and -duplication to incoming packets, before relaying them.
035 * <br>
036 * The specific emulation parameters can be tuned with the appropriate getters / setters,
037 * before {@link com.github.mucaho.jnetemu.DatagramWanEmulator#startEmulation() starting the emulation} or while emulation is already active.
038 */
039public class SimpleWanEmulator extends DatagramWanEmulator {
040
041    private final static Random random = new Random();
042
043    /**
044     * Construct a new {@code SimpleWanEmulator}. This will automatically create a new socket
045     * bound to the specified {@code emulatorAddress},
046     * with {@code maxPacketSize} set to {@link com.github.mucaho.jnetemu.DatagramWanEmulator#MINIMUM__PACKET_SIZE MINIMUM__PACKET_SIZE}.
047     *
048     * @param emulatorAddress the emulator address
049     * @param socketAddressA  one socket address
050     * @param socketAddressB  the other socket address
051     * @see DatagramWanEmulator#DatagramWanEmulator(SocketAddress, SocketAddress, SocketAddress)
052     */
053    public SimpleWanEmulator(SocketAddress emulatorAddress, SocketAddress socketAddressA, SocketAddress socketAddressB) {
054        super(emulatorAddress, socketAddressA, socketAddressB);
055    }
056
057    /**
058     * Construct a new {@code SimpleWanEmulator}. This will automatically create a new socket
059     * bound to the specified {@code emulatorAddress}.
060     *
061     * @param emulatorAddress the emulator address
062     * @param socketAddressA  one socket address
063     * @param socketAddressB  the other socket address
064     * @param maxPacketSize   the maximum size of incoming / outgoing datagram packets (in bytes).
065     * @see DatagramWanEmulator#DatagramWanEmulator(SocketAddress, SocketAddress, SocketAddress, int)
066     */
067    public SimpleWanEmulator(SocketAddress emulatorAddress, SocketAddress socketAddressA, SocketAddress socketAddressB, int maxPacketSize) {
068        super(emulatorAddress, socketAddressA, socketAddressB, maxPacketSize);
069    }
070
071
072    //////////////////
073    /// PROPERTIES ///
074    //////////////////
075
076    /**
077     * The chance of a packet being lost. A value of {@code 0.00f} means no packet loss,
078     * a value of {@code 1.00f} means every packet will be lost.
079     * Defaults to {@code 0.10f} ({@code 10%}).
080     */
081    private volatile float loss = 0.1f;
082
083    /**
084     * The chance of a packet being duplicated. A value of {@code 0.00f} means no
085     * packets will be duplicated, a value of {@code 1.00f} means every packet will
086     * be duplicated infinitely.
087     * Defaults to {@code 0.03f} ({@code 3%}).
088     */
089    private volatile float duplication = 0.03f;
090
091    /**
092     * The base delay between sending from one peer to receiveing on the other one.
093     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
094     * Defaults to {@code 175 ms}.
095     */
096    private volatile int delay = 175;
097
098    /**
099     * The delay jitter between sending from one peer to receiveing on the other one.
100     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
101     * Defaults to {@code +/- 75 ms}.
102     */
103    private volatile int jitter = 75;
104
105    /**
106     * Get the chance of a packet being lost. A value of {@code 0.00f} means no packet loss,
107     * a value of {@code 1.00f} means every packet will be lost.
108     * Defaults to {@code 0.10f} ({@code 10%}).
109     *
110     * @return the packet loss fraction
111     */
112    public float loss() {
113        return loss;
114    }
115
116    /**
117     * Set the chance of a packet being lost. A value of {@code 0.0f} means no packet loss,
118     * a value of {@code 1.0f} means every packet will be lost.
119     * Defaults to {@code 0.10f} ({@code 10%}).
120     *
121     * @param loss the new packet loss fraction
122     */
123    public void loss(float loss) {
124        this.loss = loss;
125    }
126
127    /**
128     * Get the chance of a packet being duplicated. A value of {@code 0.00f} means no
129     * packets will be duplicated, a value of {@code 1.00f} means every packet will
130     * be duplicated infinitely.
131     * Defaults to {@code 0.03f} ({@code 3%}).
132     *
133     * @return the packet duplication fraction
134     */
135    public float duplication() {
136        return duplication;
137    }
138
139    /**
140     * Set the chance of a packet being duplicated. A value of {@code 0.00f} means no
141     * packets will be duplicated, a value of {@code 1.00f} means every packet will
142     * be duplicated infinitely.
143     * Defaults to {@code 0.03f} ({@code 3%}).
144     *
145     * @param duplication the new packet duplication fraction
146     */
147    public void duplication(float duplication) {
148        this.duplication = duplication;
149    }
150
151
152    /**
153     * Get the base delay between sending from one peer to receiveing on the other one.
154     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
155     * Defaults to {@code 175 ms}.
156     *
157     * @return the base delay in ms
158     */
159    public int delay() {
160        return delay;
161    }
162
163    /**
164     * Set the base delay between sending from one peer to receiveing on the other one.
165     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
166     * Defaults to {@code 175 ms}.
167     *
168     * @param delay the new base delay in ms
169     */
170    public void delay(int delay) {
171        this.delay = delay;
172    }
173
174    /**
175     * Get the delay jitter between sending from one peer to receiveing on the other one.
176     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
177     * Defaults to {@code +/- 75 ms}.
178     *
179     * @return the delay jitter in ms
180     */
181    public int jitter() {
182        return jitter;
183    }
184
185    /**
186     * Set the delay jitter between sending from one peer to receiveing on the other one.
187     * The actual delay will randomly vary between {@code baseDelay +/- jitter}.
188     * Defaults to {@code +/- 75 ms}.
189     *
190     * @param jitter the new delay jitter in ms
191     */
192    public void jitter(int jitter) {
193        this.jitter = jitter;
194    }
195
196
197    /** {@inheritDoc} */
198    @Override
199    protected void computeNetworkConditions(long timeNow, Collection<? extends Scheduled> scheduled, List<Long> scheduledTimes) {
200        do {
201            if (random.nextFloat() >= loss)
202                scheduledTimes.add(timeNow + delay - jitter + random.nextInt(jitter * 2 + 1));
203        } while (random.nextFloat() < duplication);
204    }
205}
